		function signIn(username,password,onSuccess,onFailure){
			document.getElementById('view-poc-button').setAttribute('disabled', '');
			document.getElementById('loading-blocks').style.visibility = 'visible';

			
			var userData = {
				Username : username,
				Pool : getCurrentUserPool()
			};

			var authenticationData = {
				Username : username, 
				Password : password, 
			};

			var authenticationDetails = new AmazonCognitoIdentity.AuthenticationDetails(authenticationData);
			 
			var cognitoUser = new AmazonCognitoIdentity.CognitoUser(userData);
			
			cognitoUser.authenticateUser(authenticationDetails, {
				onSuccess: onSuccess,
				onFailure: onFailure
			});

		}		
		function signinFailure(err){
			document.getElementById('view-poc-button').removeAttribute('disabled');
			document.getElementById('loading-blocks').style.visibility = 'hidden';
			//alert('Uh Oh, something went wrong...');
			//alert(JSON.stringify(err));
			document.getElementById('bugpoc-password').style.border = '1px solid red';
			document.getElementById('bugpoc-password').style.color='red';
			animateCSS('#bugpoc-password','shake',function(){
				document.getElementById('bugpoc-password').style.border = '1px solid var(--off-black-color)';
				document.getElementById('bugpoc-password').style.color='black';
			});
		}
		function animateCSS(element, animationName, callback) {
			const node = document.querySelector(element)
			node.classList.add('animated', animationName)

			function handleAnimationEnd() {
				node.classList.remove('animated', animationName)
				node.removeEventListener('animationend', handleAnimationEnd)

				if (typeof callback === 'function') callback()
			}

			node.addEventListener('animationend', handleAnimationEnd)
		}
		
		function signinSuccess(){
			document.getElementById('view-poc-button').removeAttribute('disabled');
			document.getElementById('loading-blocks').style.visibility = 'hidden';
			animateCSS('#view-h2','fadeOut',function(){
				document.getElementById('view-h2').style.display='none';
			})
			animateCSS('#switch-toggle','fadeOut',function(){
				document.getElementById('switch-toggle').style.display='none';
			})
			animateCSS('#sign-in-wrapper','bounceOutUp',function(){
				document.getElementById('sign-in-wrapper').style.display='none';
				location.assign('/account');
			})
			
			
		}
		
		function doSignin(){
			email = document.getElementById('bugpoc-email').value;
			password = document.getElementById('bugpoc-password').value;
			signIn(email,password,signinSuccess,signinFailure);
		}
		
		setTimeout(function(){document.getElementById("bugpoc-email").focus()},1000);

		function toggleTesterOrg(isTester){
			if (isTester){
				document.body.classList.remove('bg-2');
				void document.body.offsetWidth;
				document.body.classList.add('bg');
				//document.getElementById('invite-only-div').style.display = 'none';
				//document.getElementById('join-link-div').style.display = 'inline-block';
				setCurrentUserType('tester');
			} else {
				document.body.classList.remove('bg');
				void document.body.offsetWidth;
				document.body.classList.add('bg-2');
				//document.getElementById('invite-only-div').style.display = 'inline-block';
				//document.getElementById('join-link-div').style.display = 'none';
				setCurrentUserType('organization')
			}
		}
		
		
		function populateFromFragment(){
			fragment = decodeURIComponent(window.location.hash.substr(1));
			if (fragment == ''){
				toggleTesterOrg(true);
				return
			}
			// remove hash from URI (including # sign)
			window.history.replaceState('', document.title, window.location.pathname+ window.location.search);
			if (fragment == 'org'){
				document.getElementById('q2').click();
			} 
		}
